Complete Example

<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <title>ProTable CDN Demo</title>
  <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/ant-design-vue@4/dist/antd.css">
</head>
<body>
  <div id="app"></div>

  <script src="https://cdn.jsdelivr.net/npm/vue@3/dist/vue.global.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/ant-design-vue@4/dist/antd.js"></script>
  <script src="https://cdn.jsdelivr.net/npm/vue3-pro-table-antd/dist/pro-table.umd.js"></script>

  <script>
    const { createApp } = Vue;
    const { ProTable } = vue3ProTableAntd;

    const app = createApp({
      template: `
        <ProTable
          :columns="columns"
          :request="fetchData"
          :show-search="true"
        />
      `,
      setup() {
        const columns = [
          { title: 'ID', dataIndex: 'id', name: 'id' },
          { title: 'Name', dataIndex: 'name', name: 'name' },
          { title: 'Age', dataIndex: 'age', name: 'age' }
        ];

        const fetchData = async (params) => {
          console.log('Fetching with:', params);
          // Simulate API call
          await new Promise(resolve => setTimeout(resolve, 500));

          return {
            success: true,
            data: [
              { id: 1, name: 'John', age: 25 },
              { id: 2, name: 'Jane', age: 30 }
            ],
            total: 2
          };
        };

        return { columns, fetchData };
      }
    });

    app.use(antd);
    app.mount('#app');
  </script>
</body>
</html>
